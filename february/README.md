## Задание 1. Создание потока

Напишите программу, которая создает поток. Используйте атрибуты по умолчанию. Родительский и вновь созданный поток должны распечатать десять строк текста.

**Файл с кодом:** [task1_create.c](task1_create.c)

<details>
<summary>Показать код</summary>

```c
 
#include <stdio.h>
#ifdef _WIN32
#include <windows.h>   /* для русских букв в консоли Windows */
#endif
#include <pthread.h>    /* pthread_create, pthread_join */

/* Функция, которую будет выполнять дочерний поток. */
void *child_routine(void *arg)
{
    int i;
    for (i = 1; i <= 10; i++) {
        printf("  [Дочерний поток]  строка %d\n", i);
    }
    return NULL;    /* поток завершается */
}

int main(void)
{
#ifdef _WIN32
    SetConsoleOutputCP(65001);  /* выводить русский текст (UTF-8) */
#endif
    pthread_t child;    /* идентификатор дочернего потока */
    int i;

    /* Создаём поток: атрибуты по умолчанию (NULL),
       функция child_routine, аргументов не передаём (NULL). */
    if (pthread_create(&child, NULL, child_routine, NULL) != 0) {
        printf("Ошибка: не удалось создать поток\n");
        return 1;
    }

    /* Родительский поток тоже печатает десять строк.
       Он работает ПАРАЛЛЕЛЬНО с дочерним, поэтому строки
       в выводе могут перемешаться - это нормально. */
    for (i = 1; i <= 10; i++) {
        printf("[Родительский поток] строка %d\n", i);
    }

    /* Дожидаемся завершения дочернего потока, чтобы программа
       не закончилась раньше, чем он допечатает. */
    pthread_join(child, NULL);

    return 0;
}
```

## Задание 2. Ожидание потока

Модифицируйте программу упр. 1 так, чтобы вывод родительского потока производился после завершения дочернего. Используйте pthread_join.

**Файл с кодом:** [task2_join.c](task2_join.c)

<details>
<summary>Показать код</summary>

```c
 

#include <stdio.h>
#ifdef _WIN32
#include <windows.h>   /* для русских букв в консоли Windows */
#endif
#include <pthread.h>

void *child_routine(void *arg)
{
    int i;
    for (i = 1; i <= 10; i++) {
        printf("  [Дочерний поток]  строка %d\n", i);
    }
    return NULL;
}

int main(void)
{
#ifdef _WIN32
    SetConsoleOutputCP(65001);  /* выводить русский текст (UTF-8) */
#endif
    pthread_t child;
    int i;

    if (pthread_create(&child, NULL, child_routine, NULL) != 0) {
        printf("Ошибка: не удалось создать поток\n");
        return 1;
    }

    /* КЛЮЧЕВОЕ ОТЛИЧИЕ ОТ ЗАДАНИЯ 1:

    return 0;
}
```

## Задание 3. Параметры потока

Напишите программу, которая создает четыре потока, исполняющие одну и ту же функцию. Эта функция должна распечатать последовательность текстовых строк, переданных как параметр. Каждый из созданных потоков должен распечатать различные последовательности строк.

**Файл с кодом:** [task3_params.c](task3_params.c)

<details>
<summary>Показать код</summary>

```c
 
#include <stdio.h>
#ifdef _WIN32
#include <windows.h>   /* для русских букв в консоли Windows */
#endif
#include <pthread.h>

#define NUM_THREADS 4

/* Структура-параметр: что именно поток должен печатать. */
typedef struct {
    int id;                 /* номер потока (для наглядности) */
    const char *lines[3];   /* последовательность из трёх строк */
} ThreadParam;

/* Одна общая функция для всех потоков. Получает указатель
   на ThreadParam и печатает переданные ей строки. */
void *print_routine(void *arg)
{
    ThreadParam *p = (ThreadParam *)arg;
    int i;
    for (i = 0; i < 3; i++) {
        printf("Поток %d: %s\n", p->id, p->lines[i]);
    }
    return NULL;
}

int main(void)
{
#ifdef _WIN32
    SetConsoleOutputCP(65001);  /* выводить русский текст (UTF-8) */
#endif
    pthread_t threads[NUM_THREADS];

    /* У каждого потока - свой набор строк. */
    ThreadParam params[NUM_THREADS] = {
        {1, {"яблоко",  "груша",   "слива"}},
        {2, {"красный", "зелёный", "синий"}},
        {3, {"понедельник", "вторник", "среда"}},
        {4, {"один",    "два",     "три"}}
    };

    int i;

    /* Запускаем четыре потока, каждому передаём адрес его параметров. */
    for (i = 0; i < NUM_THREADS; i++) {
        pthread_create(&threads[i], NULL, print_routine, &params[i]);
    }

    /* Ждём завершения всех четырёх. */
    for (i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    return 0;
}
```

## Задание 4. Принудительное завершение

Дочерний поток должен распечатывать текст на экран. Через две секунды после создания дочернего потока, родительский поток должен прервать ее вызовом функции pthread_cancel.

**Файл с кодом:** [task4_cancel.c](task4_cancel.c)

<details>
<summary>Показать код</summary>

```c
 

#include <stdio.h>
#ifdef _WIN32
#include <windows.h>   /* для русских букв в консоли Windows */
#endif
#include <pthread.h>
#include <unistd.h>     /* sleep, usleep */

/* Общий флаг остановки. volatile - чтобы поток всегда читал
   свежее значение, а не закэшированное. */
volatile int stop = 0;

void *child_routine(void *arg)
{
    int counter = 0;
    /* Поток работает, пока родитель не попросит остановиться. */
    while (!stop) {
        printf("  [Дочерний поток] работаю... сообщение %d\n", ++counter);
        fflush(stdout);     /* сразу выводим на экран */
        usleep(300000);     /* пауза 0.3 секунды */
    }
    return NULL;
}

int main(void)
{
#ifdef _WIN32
    SetConsoleOutputCP(65001);  /* выводить русский текст (UTF-8) */
#endif
    pthread_t child;

    if (pthread_create(&child, NULL, child_routine, NULL) != 0) {
        printf("Ошибка: не удалось создать поток\n");
        return 1;
    }

    printf("[Родитель] дочерний поток запущен, ждём 2 секунды...\n");
    fflush(stdout);
    sleep(2);   /* даём дочернему поработать 2 секунды */

    /* Просим дочерний поток остановиться. */
    stop = 1;
    printf("[Родитель] подан сигнал остановки дочернему потоку\n");
    fflush(stdout);

    /* Дожидаемся фактического завершения. */
    pthread_join(child, NULL);
    printf("[Родитель] дочерний поток завершён, выходим.\n");

    return 0;
}
```

## Задание 5. Обработка завершения

Модифицируйте программу упр. 4 так, чтобы дочерний поток перед завершением распечатывал сообщение об этом. Используйте pthread_cleanup_push.

**Файл с кодом:** [task5_cleanup.c](task5_cleanup.c)

<details>
<summary>Показать код</summary>

```c
 
#include <stdio.h>
#ifdef _WIN32
#include <windows.h>
#endif
#include <pthread.h>
#include <unistd.h>

volatile int stop = 0;

/* Функция "очистки": что поток делает перед самым завершением. */
void cleanup(void)
{
    printf("  [Дочерний поток] меня завершают - выполняю очистку и прощаюсь!\n");
    fflush(stdout);
}

void *child_routine(void *arg)
{
    int counter = 0;
    while (!stop) {
        printf("  [Дочерний поток] работаю... сообщение %d\n", ++counter);
        fflush(stdout);
        usleep(300000);
    }
    /* Перед выходом выполняем очистку. */
    cleanup();
    return NULL;
}

int main(void)
{
#ifdef _WIN32
    SetConsoleOutputCP(65001);
#endif
    pthread_t child;

    if (pthread_create(&child, NULL, child_routine, NULL) != 0) {
        printf("Ошибка: не удалось создать поток\n");
        return 1;
    }

    printf("[Родитель] дочерний поток запущен, ждём 2 секунды...\n");
    fflush(stdout);
    sleep(2);

    stop = 1;
    printf("[Родитель] подан сигнал остановки\n");
    fflush(stdout);

    pthread_join(child, NULL);
    printf("[Родитель] дочерний поток завершён, выходим.\n");

    return 0;
}
```
